### Solution

```javascript
// Input: string representing a map PATTERN of trees and open spaces
// Output: number representing how many trees were encountered navigating to the bottom of the map with the specified slope
// Rules/Implicit Knowledge
//   map pattern repeats horizontally as many times as needed
//   trees only grow on integer grid
//   trees represented by X
//   open spaces represented by .
//   slope: right 3, down 1
//   count trees in locations only after moving one slope's set of values (see example); don't count every tree on the slope line
//   horizontally repeating map can be simulated with using remainder when location goes beyond end of row
// Examples: see below
// Edge Cases
// Data Structure: map as array of strings
// Algorithm
//   constants
//     split map on newlines to array of strings / array of rows
//       each string represents a row of the map
//     map width (length of a row)
//     tree marker is #
//     space marker is . (may not need this)
//   initialize trees to zero
//   initialize rowIdx to zero
//   initialize colIdx to zero
//   until reaching the end of the map (rowIdx greater than map array's length)
//     advance row and col indexes by slope value
//     if col index is greater than or equal to map width
//       set col index to remainder of col index and map width
//     if current location (at row and column indexes) is a tree, increment trees
//   return trees
// Code

const map = '..##.......\n' +
            '#...#...#..\n' +
            '.#....#..#.\n' +
            '..#.#...#.#\n' +
            '.#...##..#.\n' +
            '..#.##.....\n' +
            '.#.#.#....#\n' +
            '.#........#\n' +
            '#.##...#...\n' +
            '#...##....#\n' +
            '.#..#...#.#';

let advancePosition = function advancePosition(rowIdx, colIdx, mapWidth) {
  const SLOPE_X = 3;
  const SLOPE_Y = 1;

  colIdx += SLOPE_X;
  rowIdx += SLOPE_Y;

  if (colIdx >= mapWidth) colIdx %= mapWidth;

  return [rowIdx, colIdx];
};

let countTrees = function countTrees(map) {
  map = map.split('\n');
  const mapWidth = map[0].length;
  const TREE_MARKER = '#';

  let trees = 0;
  let rowIdx = 0;
  let colIdx = 0;

  while (rowIdx < map.length) {
    if (map[rowIdx][colIdx] === TREE_MARKER) trees += 1;
    [rowIdx, colIdx] = advancePosition(rowIdx, colIdx, mapWidth);
  }

  return trees;
};

console.log(countTrees(map) === 7);
```

### Part 2

Time to check the rest of the slopes - you need to minimize the probability of a sudden arboreal stop, after all.

Determine the number of trees you would encounter if, for each of the following slopes, you start at the top-left corner and traverse the map all the way to the bottom:

* Right 1, down 1.
* Right 3, down 1. (This is the slope you already checked.)
* Right 5, down 1.
* Right 7, down 1.
* Right 1, down 2.

In the above example, these slopes would find `2`, `7`, `3`, `4`, and `2` tree(s) respectively; multiplied together, these produce the answer `336`.

**What do you get if you multiply together the number of trees encountered on each of the listed slopes?**

### Solution

4385176320

```javascript
// Slope just needs to be a variable instead of hardcoded.
// The list of slopes can be provided as an array.
// Map the array of slopes to the number of trees encountered.
// Reduce the array of trees by multiplication.

const slopes = [[1, 1], [3, 1], [5, 1], [7, 1], [1,2]];

let advancePosition = function advancePosition(slope, rowIdx, colIdx, mapWidth) {
  // const SLOPE_X = 3;
  // const SLOPE_Y = 1;

  const [SLOPE_X, SLOPE_Y] = slope;

  colIdx += SLOPE_X;
  rowIdx += SLOPE_Y;

  if (colIdx >= mapWidth) colIdx %= mapWidth;

  return [rowIdx, colIdx];
};

let countTrees = function countTrees(map, slope) {
  map = map.split('\n');
  const mapWidth = map[0].length;
  const TREE_MARKER = '#';

  let trees = 0;
  let rowIdx = 0;
  let colIdx = 0;

  while (rowIdx < map.length) {
    if (map[rowIdx][colIdx] === TREE_MARKER) trees += 1;
    [rowIdx, colIdx] = advancePosition(slope, rowIdx, colIdx, mapWidth);
  }

  return trees;
};

let multiplyTrees = function multiplyTrees(map, slopes) {
  let treeCounts = slopes.map(slope => countTrees(map, slope));
  return treeCounts.reduce((product, count) => product * count, 1);
};
```
